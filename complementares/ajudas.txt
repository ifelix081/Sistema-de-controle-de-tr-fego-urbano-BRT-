#include <stdio.h>
#include <stdlib.h>

struct BRT {
    int linha;
    int numero_onibus;
    char garagem[50];
};

// ðŸ”¹ FunÃ§Ã£o para cadastrar
void cadastrar() {
    FILE *arquivo = fopen("dados.txt", "a");
    struct BRT b;

    if (arquivo == NULL) {
        printf("Erro ao abrir arquivo!\n");
        return;
    }

    printf("\n--- Cadastro ---\n");
    printf("Linha: ");
    scanf("%d", &b.linha);

    printf("Numero do onibus: ");
    scanf("%d", &b.numero_onibus);

    printf("Garagem: ");
    scanf("%s", b.garagem);

    fprintf(arquivo, "%d %d %s\n", b.linha, b.numero_onibus, b.garagem);

    fclose(arquivo);

    printf("Cadastro realizado com sucesso!\n");
}

// ðŸ”¹ FunÃ§Ã£o para listar dados
void listar() {
    FILE *arquivo = fopen("dados.txt", "r");
    struct BRT b;

    if (arquivo == NULL) {
        printf("Nenhum dado encontrado.\n");
        return;
    }

    printf("\n--- Lista de Onibus ---\n");

    while (fscanf(arquivo, "%d %d %s", &b.linha, &b.numero_onibus, b.garagem) != EOF) {
        printf("Linha: %d | Onibus: %d | Garagem: %s\n",
               b.linha, b.numero_onibus, b.garagem);
    }

    fclose(arquivo);
}

// ðŸ”¹ FunÃ§Ã£o de relatÃ³rio
void relatorio() {
    FILE *arquivo = fopen("dados.txt", "r");
    struct BRT b;

    int total = 0;

    if (arquivo == NULL) {
        printf("Nenhum dado para relatorio.\n");
        return;
    }

    while (fscanf(arquivo, "%d %d %s", &b.linha, &b.numero_onibus, b.garagem) != EOF) {
        total++;
    }

    fclose(arquivo);

    FILE *rel = fopen("relatorio.txt", "w");

    fprintf(rel, "RELATORIO BRT\n");
    fprintf(rel, "Total de onibus: %d\n", total);

    fclose(rel);

    printf("\nRelatorio gerado com sucesso!\n");
}

// ðŸ”¹ Menu principal
int main() {
    int opcao;

    do {
        printf("\n=== SISTEMA BRT ===\n");
        printf("1 - Cadastrar\n");
        printf("2 - Listar\n");
        printf("3 - Gerar Relatorio\n");
        printf("0 - Sair\n");
        printf("Escolha: ");
        scanf("%d", &opcao);

        switch(opcao) {
            case 1:
                cadastrar();
                break;
            case 2:
                listar();
                break;
            case 3:
                relatorio();
                break;
            case 0:
                printf("Saindo...\n");
                break;
            default:
                printf("Opcao invalida!\n");
        }

    } while(opcao != 0);

    return 0;
}
